201/**
202 * @brief Aggiorna lo stato di un encoder
203 *
204 * @param e_x Puntatore alla struttura dell'encoder da aggiornare
205 *
206 * @details Questa funzione aggiorna la velocità, la posizione e gestisce
207 * la saturazione per un encoder. Implementa un modello di movimento
208 * basato su accelerazione e velocità, considerando lo sfasamento tra
209 * i canali A e B dell'encoder.
210 *
211 * Il processo di aggiornamento include:
212 * 1. Integrazione dell'accelerazione per ottenere la velocità
213 * 2. Saturazione della velocità entro i limiti da VELOCITA_MAX
214 * 3. Integrazione della velocità per ottenere la posizione
215 * 4. Calcolo della posizione per entrambi i canali A e B
216 * 5. Gestione della saturazione dello spazio
217 *
218 * @note
219 * - La variabile t_update rappresenta l'intervallo di tempo per
220 * l'integrazione
221 *
222 * @see encoder
223 * @see VELOCITA_MAX
224 * @see t_update
225 */
226static void aggiorna_encoder(encoder *e_x)
227{
228 /*
229 * Contiene la posizione MINORE tra quella
230 * del canale A e B, serve per la correzione
231 * dello spazio
232 */
233 double_t pos_minore;
234
235 /*
236 * Contiene la posizione MAGGIORE tra quella
237 * del canale A e B, serve per la correzione
238 * dello spazio
239 */
240 double_t pos_maggiore;
241
242 /* Integrazione dell'accelerazione */
243 e_x->vel = (e_x->acc * t_update) + e_x->vel;
244
245 /* Saturo la velocità se va oltre la soglia fissata */
246 if (e_x->vel > VELOCITA_MAX)
247 {
248 e_x->vel = VELOCITA_MAX;
249 }
250 else if (e_x->vel < -VELOCITA_MAX)
251 {
252 e_x->vel = -VELOCITA_MAX;
253 }
254 else
255 {
256 /* Non succede niente, MISRA-2023-15.7 */
257 }
258
259 /*
260 * Integro la velocità, assegno lo spazio ad A
261 * (è una scelta arbitraria)
262 */
263 e_x->pos_A = e_x->pos_A + (e_x->vel * t_update);
264
265 /*
266 * Estrapolo il fattore di sfasamento, cio� converto i gradi nella
267 * quantit� di spazio da cui il canale B si discosta dal canale A
268 */
269 float_t k_fase = ((float_t) - (e_x->fase) / 360);
270 e_x->pos_B = e_x->pos_A + (2 * e_x->l_passo * k_fase);
271
272 /*
273 * Indico chi è il sensore con la posizione maggiore e quale con la
274 * minore, mi serve per fare il controllo sulla correzione di spazio
275 */
276 if (k_fase <= 0)
277 {
278 /* Posizione del B è minore dell'A */
279 pos_minore = e_x->pos_B;
280 pos_maggiore = e_x->pos_A;
281 }
282 else
283 {
284 /* Posizione dell'A è minore del B */
285 pos_minore = e_x->pos_A;
286 pos_maggiore = e_x->pos_B;
287 }
288
289 /* Satura lo spazio se uno dei due sensori va oltre la
290 soglia [-2*passi, 2*passi] */
291 if (pos_minore <= (-2 * e_x->l_passo))
292 {
293 /*
294 * Il MINORE ha sforato, esso ritorna a 0 mentre il maggiore
295 * va ad un valore maggiore di 0 (dipende dalla fase)
296 */
297 e_x->pos_A = e_x->pos_A + (2 * e_x->l_passo);
298 e_x->pos_B = e_x->pos_B + (2 * e_x->l_passo);
299 }
300 else if (pos_maggiore >= (2 * e_x->l_passo))
301 {
302 /*
303 * Il MAGGIORE ha sforato, esso ritorna a 0 mentre il minore
304 * va ad un valore minore di 0 (dipende dalla fase)
305 */
306 e_x->pos_A = e_x->pos_A - (2 * e_x->l_passo);
307 e_x->pos_B = e_x->pos_B - (2 * e_x->l_passo);
308 }
309 else
310 {
311 /* Non succede niente, MISRA-2023-15.7 */
312 }
313
314 return;
315}

317/**
318 * @brief Emula il comportamento di un encoder
319 *
320 * @param e_x Puntatore alla struttura dell'encoder da emulare
321 *
322 * @details
323 * Questa funzione simula il comportamento di un encoder generando segnali
324 * per i canali A e B basati sulla posizione attuale dell'encoder. Il processo
325 * include:
326 * 1. Calcolo delle soglie basate sul duty cycle per ciascun canale
327 * 2. Generazione dei segnali per i canali A e B usando GPIO
328 * 3. Valutazione dello stato dell'encoder basata sui segnali generati
329 *
330 * @note
331 * - Assume l'uso di XGpio per la scrittura dei segnali
332 * - Il comportamento è influenzato dai duty cycle e dalle posizioni dei canali
333 * A e B
334 * - Utilizza la funzione valuta_stato_encoder per aggiornare lo stato dell'
335 * encoder
336 *
337 * @see encoder, valuta_stato_encoder, XGpio_DiscreteWrite
338 */
339static void emula_encoder(encoder *e_x)
340{
341 /* Stato del canale, equivalente al valore logico di GPIO corrispondente */
342 bool stato_sensoreA;
343 bool stato_sensoreB;
344
345 /* Porto i duty cycle da percentuale intera a decimale */
346 float_t k_dutyA = ((float_t) e_x->duty_A) * 0.01;
347 float_t k_dutyB = ((float_t) e_x->duty_B) * 0.01;
348
349 /*
350 * Genero le soglie su cui fare il confronto, per stimare in che
351 * posizione è il sensore
352 */
353 double_t soglia_max_def = 2 * e_x->l_passo; /* 2passi */
354 double_t soglia_min_def = - soglia_max_def; /* -2passi */
355 /* Duty% * 2passi */
356 double_t soglia_pos_A = k_dutyA * soglia_max_def;
357 double_t soglia_pos_B = k_dutyB * soglia_max_def;
358 /* (1-Duty%) * 2passi */
359 double_t soglia_neg_A = (1 - k_dutyA) * soglia_min_def;
360 double_t soglia_neg_B = (1 - k_dutyB) * soglia_min_def;
361
362 /*
363 * Generazione segnale per canale A da GPIO,
364 * se il treno va a velocit� positiva allora lo spazio sar� positivo,
365 * quindi verranno usati solo gli ultimi 2 if dei 4 totali, che fanno
366 * il check per posizioni positive. Ogni paio di if dice se la GPIO è
367 * alta o bassa, la soglia che decide questo dipende dal duty cycle.
368 * Esempio: il range dello spazio � 0 -> 2 passi, se duty = 50%
369 * la soglia di transizione alto/basso sar� 1 passo.
370 */
371 if ((e_x->pos_A >= soglia_min_def) && (e_x->pos_A < soglia_neg_A))
372 {
373 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x01);
374 stato_sensoreA = true;
375 }
376 else if ((e_x->pos_A >= soglia_neg_A) && (e_x->pos_A < 0))
377 {
378 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x00);
379 stato_sensoreA = false;
380 }
381 else if ((e_x->pos_A >= 0) && (e_x->pos_A < soglia_pos_A))
382 {
383 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x01);
384 stato_sensoreA = true;
385 }
386 else if ((e_x->pos_A >= soglia_pos_A) && (e_x->pos_A < soglia_max_def))
387 {
388 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x00);
389 stato_sensoreA = false;
390 }
391 else
392 {
393 /* Non succede niente, MISRA-2023-15.7 */
394 }
395
396 /*
397 * Generazione segnale per canale B da GPIO, valgono gli stessi
398 * ragionamenti del canale A
399 */
400 if ((e_x->pos_B >= soglia_min_def) && (e_x->pos_B < soglia_neg_B))
401 {
402 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x01);
403 stato_sensoreB = true;
404 }
405 else if ((e_x->pos_B >= soglia_neg_B) && (e_x->pos_B < 0))
406 {
407 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x00);
408 stato_sensoreB = false;
409 }
410 else if ((e_x->pos_B >= 0) && (e_x->pos_B < soglia_pos_B))
411 {
412 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x01);
413 stato_sensoreB = true;
414 }
415 else if ((e_x->pos_B >= soglia_pos_B) && (e_x->pos_B < soglia_max_def))
416 {
417 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x00);
418 stato_sensoreB = false;
419 }
420 else
421 {
422 /* Non succede niente, MISRA-2023-15.7 */
423 }
424
425 valuta_stato_encoder(e_x, stato_sensoreA, stato_sensoreB);
426}

428/**
429 * @brief Inizializza una struttura encoder con valori predefiniti
430 *
431 * @param e_x Puntatore alla struttura dell'encoder da inizializzare
432 *
433 * @details
434 * Imposta i valori iniziali per tutti i campi della struttura encoder,
435 * inclusi:
436 * velocità, accelerazione, posizioni, duty cycle, fase, conteggio, stato di
437 * incollaggio,
438 * impulsi per rivoluzione (ppr), diametro e lunghezza del passo.
439 *
440 * @note
441 * - Il valore di PI_GRECO deve essere definito altrove nel codice
442 * - I valori iniziali sono pensati per un encoder standard a quadratura
443 *
444 * @see encoder
445 */

447{
448 e_x->vel = 0;
449 e_x->acc = 0;
450 e_x->pos_A = 0;
451 e_x->pos_B = PI_GRECO / 512;
452 e_x->duty_A = 50;
453 e_x->duty_B = 50;
454 e_x->fase = 90;
455 e_x->conteggio = 0;
456 e_x->incollaggio_A = false;
457 e_x->incollaggio_B = false;
458 e_x->ppr = 128;
459 e_x->diametro = 1;
460 e_x->l_passo = PI_GRECO / 256;

463/**
464 * @brief Resetta lo stato dei GPIO associati all'encoder
465 *
466 * @param e_x Puntatore alla struttura dell'encoder
467 *
468 * @details
469 * Configura i GPIO associati ai canali A e B dell'encoder come output

488/**
489 * @brief Valuta e aggiorna lo stato dell'encoder basato sui segnali dei canali
490 * A e B
491 *
492 * @param e_x Puntatore alla struttura dell'encoder
493 * @param statoA Stato logico del canale A
494 * @param statoB Stato logico del canale B
495 *
496 * @details
497 * Questa funzione determina lo stato dell'encoder basandosi sui segnali dei
498 * canali A e B.
499 * Aggiorna il conteggio dell'encoder quando viene rilevato un cambiamento di
500 * stato valido.
501 * Gli stati possibili sono: zero, uno, due, tre e incerto.
502 *
503 * @note
504 * - Il conteggio viene incrementato solo quando lo stato cambia da uno stato
505 * valido a un altro
506 * - Lo stato 'incerto' è usato per gestire l'inizializzazione
507 *
508 * @see encoder
509 */
510static void valuta_stato_encoder(encoder *e_x, bool statoA, bool statoB)
511{
512
513 if((statoA == false) && (statoB == false) && (e_x->stato != zero))
514 {
515 if(e_x->stato != incerto)
516 {
517 e_x->conteggio++;
518 }
519 else
520 {
521 /* Non succede niente, MISRA-2023-15.7 */
522 }
523 e_x->stato = zero;
524 }
525 else if((statoA == false) && (statoB == true) && (e_x->stato != uno))
526 {
527 if(e_x->stato != incerto)
528 {
529 e_x->conteggio++;
530 }
531 else
532 {
533 /* Non succede niente, MISRA-2023-15.7 */
534 }
535 e_x->stato = uno;
536 }
537 else if((statoA == true) && (statoB == false) && (e_x->stato != due))
538 {
539 if(e_x->stato != incerto)
540 {
541 e_x->conteggio++;
542 }
543 else
544 {
545 /* Non succede niente, MISRA-2023-15.7 */
546 }
547 e_x->stato = due;
548 }
549 else if((statoA == true) && (statoB == true) && (e_x->stato != tre))
550 {
551 if(e_x->stato != incerto)
552 {
553 e_x->conteggio++;
554 }
555 else
556 {
557 /* Non succede niente, MISRA-2023-15.7 */
558 }
559 e_x->stato = tre;
560 }
561 else
562 {
563 /* Non succede niente, MISRA-2023-15.7 */
564 }

566}
