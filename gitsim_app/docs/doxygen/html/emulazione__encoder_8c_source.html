62/**
63 * @brief Aggiorna lo stato di un encoder
64 *
65 * @param e_x Puntatore alla struttura dell'encoder da aggiornare
66 *
67 * @details Questa funzione aggiorna la velocità, la posizione e gestisce
68 * la saturazione per un encoder. Implementa un modello di movimento
69 * basato su accelerazione e velocità, considerando lo sfasamento tra
70 * i canali A e B dell'encoder.
71 *
72 * Il processo di aggiornamento include:
73 * 1. Integrazione dell'accelerazione per ottenere la velocità
74 * 2. Saturazione della velocità entro i limiti da VELOCITA_MAX
75 * 3. Integrazione della velocità per ottenere la posizione
76 * 4. Calcolo della posizione per entrambi i canali A e B
77 * 5. Gestione della saturazione dello spazio
78 *
79 * @note
80 * - La variabile t_update rappresenta l'intervallo di tempo per
81 * l'integrazione
82 *
83 * @see encoder
84 * @see VELOCITA_MAX
85 * @see t_update
86 */
87static void aggiorna_encoder(encoder *e_x)
88{
89 /*
90 * Contiene la posizione MINORE tra quella
91 * del canale A e B, serve per la correzione
92 * dello spazio
93 */
94 double_t pos_minore;
95
96 /*
97 * Contiene la posizione MAGGIORE tra quella
98 * del canale A e B, serve per la correzione
99 * dello spazio
100 */
101 double_t pos_maggiore;
102
103 /* Integrazione dell'accelerazione */
104 e_x->vel = (e_x->acc * t_update) + e_x->vel;
105
106 /* Saturo la velocità se va oltre la soglia fissata */
107 if (e_x->vel > VELOCITA_MAX)
108 {
109 e_x->vel = VELOCITA_MAX;
110 }
111 else if (e_x->vel < -VELOCITA_MAX)
112 {
113 e_x->vel = -VELOCITA_MAX;
114 }
115 else
116 {
117 /* Non succede niente, MISRA-2023-15.7 */
118 }
119
120 /*
121 * Integro la velocità, assegno lo spazio ad A
122 * (è una scelta arbitraria)
123 */
124 e_x->pos_A = e_x->pos_A + (e_x->vel * t_update);
125
126 /*
127 * Estrapolo il fattore di sfasamento, cio� converto i gradi nella
128 * quantit� di spazio da cui il canale B si discosta dal canale A
129 */
130 float_t k_fase = ((float_t) - (e_x->fase) / 360);
131 e_x->pos_B = e_x->pos_A + (2 * e_x->l_passo * k_fase);
132
133 /*
134 * Indico chi è il sensore con la posizione maggiore e quale con la
135 * minore, mi serve per fare il controllo sulla correzione di spazio
136 */
137 if (k_fase <= 0)
138 {
139 /* Posizione del B è minore dell'A */
140 pos_minore = e_x->pos_B;
141 pos_maggiore = e_x->pos_A;
142 }
143 else
144 {
145 /* Posizione dell'A è minore del B */
146 pos_minore = e_x->pos_A;
147 pos_maggiore = e_x->pos_B;
148 }
149
150 /* Satura lo spazio se uno dei due sensori va oltre la
151 soglia [-2*passi, 2*passi] */
152 if (pos_minore <= (-2 * e_x->l_passo))
153 {
154 /*
155 * Il MINORE ha sforato, esso ritorna a 0 mentre il maggiore
156 * va ad un valore maggiore di 0 (dipende dalla fase)
157 */
158 e_x->pos_A = e_x->pos_A + (2 * e_x->l_passo);
159 e_x->pos_B = e_x->pos_B + (2 * e_x->l_passo);
160 }
161 else if (pos_maggiore >= (2 * e_x->l_passo))
162 {
163 /*
164 * Il MAGGIORE ha sforato, esso ritorna a 0 mentre il minore
165 * va ad un valore minore di 0 (dipende dalla fase)
166 */
167 e_x->pos_A = e_x->pos_A - (2 * e_x->l_passo);
168 e_x->pos_B = e_x->pos_B - (2 * e_x->l_passo);
169 }
170 else
171 {
172 /* Non succede niente, MISRA-2023-15.7 */
173 }
174
175 return;
176}

178/**
179 * @brief Emula il comportamento di un encoder
180 *
181 * @param e_x Puntatore alla struttura dell'encoder da emulare
182 *
183 * @details
184 * Questa funzione simula il comportamento di un encoder generando segnali
185 * per i canali A e B basati sulla posizione attuale dell'encoder. Il processo
186 * include:
187 * 1. Calcolo delle soglie basate sul duty cycle per ciascun canale
188 * 2. Generazione dei segnali per i canali A e B usando GPIO
189 * 3. Valutazione dello stato dell'encoder basata sui segnali generati
190 *
191 * @note
192 * - Assume l'uso di XGpio per la scrittura dei segnali
193 * - Il comportamento è influenzato dai duty cycle e dalle posizioni dei canali
194 * A e B
195 * - Utilizza la funzione valuta_stato_encoder per aggiornare lo stato dell'
196 * encoder
197 *
198 * @see encoder, valuta_stato_encoder, XGpio_DiscreteWrite
199 */
200static void emula_encoder(encoder *e_x)
201{
202 /* Stato del canale, equivalente al valore logico di GPIO corrispondente */
203 bool stato_sensoreA;
204 bool stato_sensoreB;
205
206 /* Porto i duty cycle da percentuale intera a decimale */
207 float_t k_dutyA = ((float_t) e_x->duty_A) * 0.01;
208 float_t k_dutyB = ((float_t) e_x->duty_B) * 0.01;
209
210 /*
211 * Genero le soglie su cui fare il confronto, per stimare in che
212 * posizione è il sensore
213 */
214 double_t soglia_max_def = 2 * e_x->l_passo; /* 2passi */
215 double_t soglia_min_def = - soglia_max_def; /* -2passi */
216 /* Duty% * 2passi */
217 double_t soglia_pos_A = k_dutyA * soglia_max_def;
218 double_t soglia_pos_B = k_dutyB * soglia_max_def;
219 /* (1-Duty%) * 2passi */
220 double_t soglia_neg_A = (1 - k_dutyA) * soglia_min_def;
221 double_t soglia_neg_B = (1 - k_dutyB) * soglia_min_def;
222
223 /*
224 * Generazione segnale per canale A da GPIO,
225 * se il treno va a velocit� positiva allora lo spazio sar� positivo,
226 * quindi verranno usati solo gli ultimi 2 if dei 4 totali, che fanno
227 * il check per posizioni positive. Ogni paio di if dice se la GPIO è
228 * alta o bassa, la soglia che decide questo dipende dal duty cycle.
229 * Esempio: il range dello spazio � 0 -> 2 passi, se duty = 50%
230 * la soglia di transizione alto/basso sar� 1 passo.
231 */
232 if ((e_x->pos_A >= soglia_min_def) && (e_x->pos_A < soglia_neg_A))
233 {
234 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x01);
235 stato_sensoreA = true;
236 }
237 else if ((e_x->pos_A >= soglia_neg_A) && (e_x->pos_A < 0))
238 {
239 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x00);
240 stato_sensoreA = false;
241 }
242 else if ((e_x->pos_A >= 0) && (e_x->pos_A < soglia_pos_A))
243 {
244 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x01);
245 stato_sensoreA = true;
246 }
247 else if ((e_x->pos_A >= soglia_pos_A) && (e_x->pos_A < soglia_max_def))
248 {
249 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x00);
250 stato_sensoreA = false;
251 }
252 else
253 {
254 /* Non succede niente, MISRA-2023-15.7 */
255 }
256
257 /*
258 * Generazione segnale per canale B da GPIO, valgono gli stessi
259 * ragionamenti del canale A
260 */
261 if ((e_x->pos_B >= soglia_min_def) && (e_x->pos_B < soglia_neg_B))
262 {
263 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x01);
264 stato_sensoreB = true;
265 }
266 else if ((e_x->pos_B >= soglia_neg_B) && (e_x->pos_B < 0))
267 {
268 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x00);
269 stato_sensoreB = false;
270 }
271 else if ((e_x->pos_B >= 0) && (e_x->pos_B < soglia_pos_B))
272 {
273 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x01);
274 stato_sensoreB = true;
275 }
276 else if ((e_x->pos_B >= soglia_pos_B) && (e_x->pos_B < soglia_max_def))
277 {
278 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x00);
279 stato_sensoreB = false;
280 }
281 else
282 {
283 /* Non succede niente, MISRA-2023-15.7 */
284 }
285
286 valuta_stato_encoder(e_x, stato_sensoreA, stato_sensoreB);
287}

289/**
290 * @brief Inizializza una struttura encoder con valori predefiniti
291 *
292 * @param e_x Puntatore alla struttura dell'encoder da inizializzare
293 *
294 * @details
295 * Imposta i valori iniziali per tutti i campi della struttura encoder,
296 * inclusi:
297 * velocità, accelerazione, posizioni, duty cycle, fase, conteggio, stato di
298 * incollaggio,
299 * impulsi per rivoluzione (ppr), diametro e lunghezza del passo.
300 *
301 * @note
302 * - Il valore di PI_GRECO deve essere definito altrove nel codice
303 * - I valori iniziali sono pensati per un encoder standard a quadratura
304 *
305 * @see encoder
306 */
307static void inizializza_encoder(encoder *e_x)
308{
309 e_x->vel = 0;
310 e_x->acc = 0;
311 e_x->pos_A = 0;
312 e_x->pos_B = 0;
313 e_x->duty_A = 50;
314 e_x->duty_B = 50;
315 e_x->fase = 90;
316 e_x->conteggio = 0;
317 e_x->incollaggio_A = false;
318 e_x->incollaggio_B = false;
319 e_x->ppr = 128;
320 e_x->diametro = 1;
321 e_x->l_passo = PI_GRECO / 256;

324/**
325 * @brief Resetta lo stato dei GPIO associati all'encoder
326 *
327 * @param e_x Puntatore alla struttura dell'encoder
328 *
329 * @details
330 * Configura i GPIO associati ai canali A e B dell'encoder come output
331 * e li imposta a un valore basso (0).
332 *
333 * @note
334 * - Utilizza le funzioni XGpio per la configurazione e il controllo dei GPIO
335 * - Assume che i campi indirizzo_gpio_A e indirizzo_gpio_B siano correttamente
336 * inizializzati
337 *
338 * @see encoder, XGpio_SetDataDirection, XGpio_DiscreteClear
339 */
340static void reset_gpio(encoder *e_x)
341{
342
343 XGpio_SetDataDirection(&e_x->indirizzo_gpio_A, 1, 0x00);
344 XGpio_DiscreteClear(&e_x->indirizzo_gpio_A, 1, 0x01);
345 XGpio_SetDataDirection(&e_x->indirizzo_gpio_B, 1, 0x00);
346 XGpio_DiscreteClear(&e_x->indirizzo_gpio_B, 1, 0x01);

349/**
350 * @brief Valuta e aggiorna lo stato dell'encoder basato sui segnali dei canali
351 * A e B
352 *
353 * @param e_x Puntatore alla struttura dell'encoder
354 * @param statoA Stato logico del canale A
355 * @param statoB Stato logico del canale B
356 *
357 * @details
358 * Questa funzione determina lo stato dell'encoder basandosi sui segnali dei
359 * canali A e B.
360 * Aggiorna il conteggio dell'encoder quando viene rilevato un cambiamento di
361 * stato valido.
362 * Gli stati possibili sono: zero, uno, due, tre e incerto.
363 *
364 * @note
365 * - Il conteggio viene incrementato solo quando lo stato cambia da uno stato
366 * valido a un altro
367 * - Lo stato 'incerto' è usato per gestire l'inizializzazione
368 *
369 * @see encoder
370 */
371static void valuta_stato_encoder(encoder *e_x, bool statoA, bool statoB)
372{
373
374 if((statoA == false) && (statoB == false) && (e_x->stato != zero))
375 {
376 if(e_x->stato != incerto)
377 {
378 e_x->conteggio++;
379 }
380 else
381 {
382 /* Non succede niente, MISRA-2023-15.7 */
383 }
384 e_x->stato = zero;
385 }
386 else if((statoA == false) && (statoB == true) && (e_x->stato != uno))
387 {
388 if(e_x->stato != incerto)
389 {
390 e_x->conteggio++;
391 }
392 else
393 {
394 /* Non succede niente, MISRA-2023-15.7 */
395 }
396 e_x->stato = uno;
397 }
398 else if((statoA == true) && (statoB == false) && (e_x->stato != due))
399 {
400 if(e_x->stato != incerto)
401 {
402 e_x->conteggio++;
403 }
404 else
405 {
406 /* Non succede niente, MISRA-2023-15.7 */
407 }
408 e_x->stato = due;
409 }
410 else if((statoA == true) && (statoB == true) && (e_x->stato != tre))
411 {
412 if(e_x->stato != incerto)
413 {
414 e_x->conteggio++;
415 }
416 else
417 {
418 /* Non succede niente, MISRA-2023-15.7 */
419 }
420 e_x->stato = tre;
421 }
422 else
423 {
424 /* Non succede niente, MISRA-2023-15.7 */
425 }
