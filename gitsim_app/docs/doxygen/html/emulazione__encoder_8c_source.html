62/**
63 * @brief Aggiorna lo stato di un encoder
64 *
65 * @param e_x Puntatore alla struttura dell'encoder da aggiornare
66 *
67 * @details Questa funzione aggiorna la velocità, la posizione e gestisce
68 * la saturazione per un encoder. Implementa un modello di movimento
69 * basato su accelerazione e velocità, considerando lo sfasamento tra
70 * i canali A e B dell'encoder.
71 *
72 * Il processo di aggiornamento include:
73 * 1. Integrazione dell'accelerazione per ottenere la velocità
74 * 2. Saturazione della velocità entro i limiti da VELOCITA_MAX
75 * 3. Integrazione della velocità per ottenere la posizione
76 * 4. Calcolo della posizione per entrambi i canali A e B
77 * 5. Gestione della saturazione dello spazio
78 *
79 * @note
80 * - La costante VELOCITA_MAX definisce il limite di saturazione della velocità.
81 * - La variabile t_update rappresenta l'intervallo di tempo per l'aggiornamento.
82 *
83 * @see encoder
84 */
85static void aggiorna_encoder(encoder *e_x)
86{
87 /*
88 * Contiene la posizione MINORE tra quella
89 * del canale A e B, serve per la correzione
90 * dello spazio
91 */
92 double_t pos_minore;
93
94 /*
95 * Contiene la posizione MAGGIORE tra quella
96 * del canale A e B, serve per la correzione
97 * dello spazio
98 */
99 double_t pos_maggiore;
100
101 /* Integrazione dell'accelerazione */
102 e_x->vel = (e_x->acc * t_update) + e_x->vel;
103
104 /* Saturo la velocità se va oltre la soglia fissata */
105 if (e_x->vel > VELOCITA_MAX)
106 {
107 e_x->vel = VELOCITA_MAX;
108 }
109 else if (e_x->vel < -VELOCITA_MAX)
110 {
111 e_x->vel = -VELOCITA_MAX;
112 }
113 else
114 {
115 /* Non succede niente, MISRA-2023-15.7 */
116 }
117
118 /*
119 * Integro la velocità, assegno lo spazio ad A
120 * (è una scelta arbitraria)
121 */
122 e_x->pos_A = e_x->pos_A + (e_x->vel * t_update);
123
124 /*
125 * Estrapolo il fattore di sfasamento, cio� converto i gradi nella
126 * quantit� di spazio da cui il canale B si discosta dal canale A
127 */
128 float_t k_fase = ((float_t) - (e_x->fase) / 360);
129 e_x->pos_B = e_x->pos_A + (2 * e_x->l_passo * k_fase);
130
131 /*
132 * Indico chi è il sensore con la posizione maggiore e quale con la
133 * minore, mi serve per fare il controllo sulla correzione di spazio
134 */
135 if (k_fase <= 0)
136 {
137 /* Posizione del B è minore dell'A */
138 pos_minore = e_x->pos_B;
139 pos_maggiore = e_x->pos_A;
140 }
141 else
142 {
143 /* Posizione dell'A è minore del B */
144 pos_minore = e_x->pos_A;
145 pos_maggiore = e_x->pos_B;
146 }
147
148 /* Satura lo spazio se uno dei due sensori va oltre la
149 soglia [-2*passi, 2*passi] */
150 if (pos_minore <= (-2 * e_x->l_passo))
151 {
152 /*
153 * Il MINORE ha sforato, esso ritorna a 0 mentre il maggiore
154 * va ad un valore maggiore di 0 (dipende dalla fase)
155 */
156 e_x->pos_A = e_x->pos_A + (2 * e_x->l_passo);
157 e_x->pos_B = e_x->pos_B + (2 * e_x->l_passo);
158 }
159 else if (pos_maggiore >= (2 * e_x->l_passo))
160 {
161 /*
162 * Il MAGGIORE ha sforato, esso ritorna a 0 mentre il minore
163 * va ad un valore minore di 0 (dipende dalla fase)
164 */
165 e_x->pos_A = e_x->pos_A - (2 * e_x->l_passo);
166 e_x->pos_B = e_x->pos_B - (2 * e_x->l_passo);
167 }
168 else
169 {
170 /* Non succede niente, MISRA-2023-15.7 */
171 }
172
173 return;
174}

176/**
177 * @brief Emula il comportamento di un encoder
178 *
179 * @param e_x Puntatore alla struttura dell'encoder da emulare
180 *
181 * @details
182 * Questa funzione simula il comportamento di un encoder generando segnali
183 * per i canali A e B basati sulla posizione attuale dell'encoder. Il processo include:
184 * 1. Calcolo delle soglie basate sul duty cycle per ciascun canale
185 * 2. Generazione dei segnali per i canali A e B usando GPIO
186 * 3. Valutazione dello stato dell'encoder basata sui segnali generati
187 *
188 * @note
189 * - Assume l'uso di XGpio per la scrittura dei segnali
190 * - Il comportamento è influenzato dai duty cycle e dalle posizioni dei canali A e B
191 * - Utilizza la funzione valuta_stato_encoder per aggiornare lo stato dell'encoder
192 *
193 * @see encoder, valuta_stato_encoder, XGpio_DiscreteWrite
194 */
195static void emula_encoder(encoder *e_x)
196{
197 /* Stato del canale, equivalente al valore logico di GPIO corrispondente */
198 bool stato_sensoreA;
199 bool stato_sensoreB;
200
201 /* Porto i duty cycle da percentuale intera a decimale */
202 float_t k_dutyA = ((float_t) e_x->duty_A) * 0.01;
203 float_t k_dutyB = ((float_t) e_x->duty_B) * 0.01;
204
205 /*
206 * Genero le soglie su cui fare il confronto, per stimare in che
207 * posizione è il sensore
208 */
209 double_t soglia_max_def = 2 * e_x->l_passo; /* 2passi */
210 double_t soglia_min_def = - soglia_max_def; /* -2passi */
211 /* Duty% * 2passi */
212 double_t soglia_pos_A = k_dutyA * soglia_max_def;
213 double_t soglia_pos_B = k_dutyB * soglia_max_def;
214 /* (1-Duty%) * 2passi */
215 double_t soglia_neg_A = (1 - k_dutyA) * soglia_min_def;
216 double_t soglia_neg_B = (1 - k_dutyB) * soglia_min_def;
217
218 /*
219 * Generazione segnale per canale A da GPIO,
220 * se il treno va a velocit� positiva allora lo spazio sar� positivo,
221 * quindi verranno usati solo gli ultimi 2 if dei 4 totali, che fanno
222 * il check per posizioni positive. Ogni paio di if dice se la GPIO �
223 * alto o basso, la soglia che decide questo dipende dal duty cycle.
224 * Esempio: il range dello spazio � 0 -> 2 passi, se duty = 50%
225 * la soglia di transizione alto/basso sar� 1 passo.
226 */
227 if ((e_x->pos_A >= soglia_min_def) && (e_x->pos_A < soglia_neg_A))
228 {
229 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x01);
230 stato_sensoreA = true;
231 }
232 else if ((e_x->pos_A >= soglia_neg_A) && (e_x->pos_A < 0))
233 {
234 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x00);
235 stato_sensoreA = false;
236 }
237 else if ((e_x->pos_A >= 0) && (e_x->pos_A < soglia_pos_A))
238 {
239 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x01);
240 stato_sensoreA = true;
241 }
242 else if ((e_x->pos_A >= soglia_pos_A) && (e_x->pos_A < soglia_max_def))
243 {
244 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_A, 1, 0x00);
245 stato_sensoreA = false;
246 }
247 else
248 {
249 /* Non succede niente, MISRA-2023-15.7 */
250 }
251
252 /*
253 * Generazione segnale per canale B da GPIO, valgono gli stessi
254 * ragionamenti del canale A
255 */
256 if ((e_x->pos_B >= soglia_min_def) && (e_x->pos_B < soglia_neg_B))
257 {
258 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x01);
259 stato_sensoreB = true;
260 }
261 else if ((e_x->pos_B >= soglia_neg_B) && (e_x->pos_B < 0))
262 {
263 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x00);
264 stato_sensoreB = false;
265 }
266 else if ((e_x->pos_B >= 0) && (e_x->pos_B < soglia_pos_B))
267 {
268 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x01);
269 stato_sensoreB = true;
270 }
271 else if ((e_x->pos_B >= soglia_pos_B) && (e_x->pos_B < soglia_max_def))
272 {
273 XGpio_DiscreteWrite(&e_x->indirizzo_gpio_B, 1, 0x00);
274 stato_sensoreB = false;
275 }
276 else
277 {
278 /* Non succede niente, MISRA-2023-15.7 */
279 }
280
281 valuta_stato_encoder(e_x, stato_sensoreA, stato_sensoreB);
282}

341/**
342 * @brief Valuta e aggiorna lo stato dell'encoder basato sui segnali dei canali A e B
343 *
344 * @param e_x Puntatore alla struttura dell'encoder
345 * @param statoA Stato logico del canale A
346 * @param statoB Stato logico del canale B
347 *
348 * @details
349 * Questa funzione determina lo stato dell'encoder basandosi sui segnali dei canali A e B.
350 * Aggiorna il conteggio dell'encoder quando viene rilevato un cambiamento di stato valido.
351 * Gli stati possibili sono: zero, uno, due, tre e incerto.
352 *
353 * @note
354 * - Il conteggio viene incrementato solo quando lo stato cambia da uno stato valido a un altro
355 * - Lo stato 'incerto' è usato per gestire l'inizializzazione
356 *
357 * @see encoder
358 */
359static void valuta_stato_encoder(encoder *e_x, bool statoA, bool statoB)
360{
361
362 if((statoA == false) && (statoB == false) && (e_x->stato != zero))
363 {
364 if(e_x->stato != incerto)
365 {
366 e_x->conteggio++;
367 }
368 else
369 {
370 /* Non succede niente, MISRA-2023-15.7 */
371 }
372 e_x->stato = zero;
373 }
374 else if((statoA == false) && (statoB == true) && (e_x->stato != uno))
375 {
376 if(e_x->stato != incerto)
377 {
378 e_x->conteggio++;
379 }
380 else
381 {
382 /* Non succede niente, MISRA-2023-15.7 */
383 }
384 e_x->stato = uno;
385 }
386 else if((statoA == true) && (statoB == false) && (e_x->stato != due))
387 {
388 if(e_x->stato != incerto)
389 {
390 e_x->conteggio++;
391 }
392 else
393 {
394 /* Non succede niente, MISRA-2023-15.7 */
395 }
396 e_x->stato = due;
397 }
398 else if((statoA == true) && (statoB == true) && (e_x->stato != tre))
399 {
400 if(e_x->stato != incerto)
401 {
402 e_x->conteggio++;
403 }
404 else
405 {
406 /* Non succede niente, MISRA-2023-15.7 */
407 }
408 e_x->stato = tre;
409 }

422/**
423 * @brief Inizializza le variabili e i GPIO per gli encoder
424 *
425 * @details

431 *
432 * @note
433 * - Utilizza funzioni esterne come ritorna_tempo_del_polling()
434 * - Utilizza costanti XPAR per l'identificazione dei dispositivi GPIO

439{

445
446 /* Inizializzo gpio */

448 XGpio_Initialize(&e_1.indirizzo_gpio_B, XPAR_AXI_GPIO_E1_B_DEVICE_ID);
449 XGpio_Initialize(&e_2.indirizzo_gpio_A, XPAR_AXI_GPIO_E2_A_DEVICE_ID);

452 /* Reset gpio */
453 reset_gpio(&e_1);
454 reset_gpio(&e_2);
