45/******************************************************************************
46 * SIDE LOOP
47 *****************************************************************************/
48void side_loop(void *CallBack_Timer)
49{
50 /* Resetto il flag di interrupt dal timer */
51 XScuTimer_ClearInterruptStatus(&istanza_timer_da_resettare);
52
53 /* Controllo se ho fatto abbastanza loop per entrane nel secondario */
54 if (counter_side_secondario == (n_loop_side_secondario - 1U))
55 {
56 /* Azioni del side loop secondario */
57 manda_telegramma_di_risposta();
58 reset_conteggi_encoder();
59 counter_side_secondario = 0;
60 }
61 else
62 {
63 counter_side_secondario++;
64 }
65
66 /* Azioni del side loop principale */
67 aggiorna_variabili_encoder();
68 emula_sensori_encoder();
69}

84}
