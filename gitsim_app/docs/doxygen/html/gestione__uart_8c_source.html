182/**
183 * @brief Legge il telegramma di connessione dall'applicazione
184 *
185 * Questa funzione riceve e processa il telegramma di connessione inviato dall'
186 * applicazione. Il telegramma contiene informazioni sul diametro della ruota
187 * e sui ppr degli encoder. La funzione verifica che questi valori rientrino
188 * nei limiti accettabili e, se validi, li assegna agli encoder e aggiorna
189 * lo stato della connessione.
190 */
191static void leggi_telegramma_di_connessione()
192{
193 uint16_t n_byte_ricevuti = 0;
194 uint8_t byte_ricevuti[L_TELEGRAMMA_CONN];
195 union float_bytes diametro;
196 union uint16_bytes ppr1;
197 union uint16_bytes ppr2;
198
199 do {
200 /* Attendo che un byte arrivi */
201 while (!(XUartPs_IsReceiveData(Uart_Ps.Config.BaseAddress)))
202 {
203 /* Wait */
204 }
205
206 /* Leggi byte ricevuto*/
207 uint8_t byte = XUartPs_ReadReg(Uart_Ps.Config.BaseAddress,
208 XUARTPS_FIFO_OFFSET) & 0xFF;
209
210 /* Salva byte ricevuto nel buffer */
211 byte_ricevuti[n_byte_ricevuti] = byte;
212 n_byte_ricevuti++;
213
214 }while(n_byte_ricevuti != L_TELEGRAMMA_CONN);
215
216
217 /* Estraggo diametro della ruota (little endian)*/
218 (void) memcpy(diametro.bytes, byte_ricevuti, sizeof(float_t));
219
220 /* Estraggo ppr encoder 1 (little endian)*/
221 (void) memcpy(ppr1.bytes, &byte_ricevuti[4], sizeof(uint16_t));
222
223 /* Estraggo ppr encoder 2 (little endian)*/
224 (void) memcpy(ppr2.bytes, &byte_ricevuti[6], sizeof(uint16_t));
225
226 /* Controllo se i parametri rientrano nei valori corretti */
227 if( (diametro.value > MAX_DIAMETRO_RUOTA) ||
228 (diametro.value < MIN_DIAMETRO_RUOTA) )
229 {
230 /* Diametro fuori dai limiti accettabili */
231 stato_connessione_app = false;
232 }
233 else if( (ppr1.value > MAX_PPR_ENCODER) ||
234 (ppr1.value < MIN_PPR_ENCODER) )
235 {
236 /* ppr 1 fuori dai limiti accettabili */
237 stato_connessione_app = false;
238 }
239 else if( (ppr2.value > MAX_PPR_ENCODER) ||
240 (ppr2.value < MIN_PPR_ENCODER) )
241 {
242 /* ppr 2 fuori dai limiti accettabili */
243 stato_connessione_app = false;
244 }
245 else
246 {
247 /* I controlli sono passati, assegno i parametri agli encoder */
248 assegna_ppr_encoder1(ppr1.value);
249 assegna_ppr_encoder2(ppr2.value);
250 assegna_diametro_ruota(diametro.value);
251 aggiorna_passo_encoder1();
252 aggiorna_passo_encoder2();
253 /* Imposta lo stato di connessione a true */
254 stato_connessione_app = true;
255 }
256
257}

259/**
260 * @brief Legge il telegramma di funzionamento dall'UART
261 *
262 * @details Questa funzione legge un telegramma di funzionamento dall'UART,
263 * estrae i dati e gli identificatori per il valore e l'addon, ed esegue
264 * l'azione corrispondente al valore ricevuto.
265 */
266static void leggi_telegramma_funzionamento()
267{
268 uint16_t n_byte_ricevuti = 0;
269 uint8_t byte_ricevuti[L_TELEGRAMMA_FUNZ];
270 uint8_t identificatore_valore;
271 uint8_t stringa_valore[L_FUNZ_VALORE];
272
273 /* TODO Aggiungere le funzioni per l'addon */
274 uint8_t stringa_addon[L_FUNZ_ADDON];
275 uint8_t identificatore_addon;
276
277
278 do {
279 /* Attendo che un byte arrivi */
280 while (!(XUartPs_IsReceiveData(Uart_Ps.Config.BaseAddress)))
281 {
282 /* Wait */
283 }
284
285 /* Leggi byte ricevuto*/
286 uint8_t byte = XUartPs_ReadReg(Uart_Ps.Config.BaseAddress,
287 XUARTPS_FIFO_OFFSET) & 0xFF;
288
289 /* Salva byte ricevuto*/
290 byte_ricevuti[n_byte_ricevuti] = byte;
291 n_byte_ricevuti++;
292
293 }while(n_byte_ricevuti != L_TELEGRAMMA_FUNZ);
294
295 /* Estraggo identificatore telegramma valore */
296 identificatore_valore = byte_ricevuti[L_FUNZ_VALORE - 1U];
297 (void) memcpy(&stringa_valore, &byte_ricevuti[0],
298 sizeof(uint8_t) * L_FUNZ_VALORE);
299 azione_funzionamento_valore(identificatore_valore, stringa_valore);
300
301
302 /* Estraggo identificatore telegramma addon */
303 identificatore_addon = byte_ricevuti[L_FUNZ_ADDON - 1U];
304 (void) memcpy(&stringa_valore, &byte_ricevuti[L_FUNZ_VALORE - 1U],
305 sizeof(uint8_t) * L_FUNZ_ADDON);
306
307
308}

310/**
311 * @brief Esegue l'azione corrispondente al valore ricevuto nel telegramma
312 *
313 * @param identificatore Byte che identifica il tipo di azione da eseguire
314 * @param array_stringa Array contenente i dati del telegramma
315 *
316 * @details Questa funzione interpreta l'identificatore ricevuto e esegue
317 * l'azione corrispondente, che può includere l'assegnazione di velocità o
318 * accelerazione agli encoder, la disconnessione dall'applicazione o il reset
319 * della cinematica degli encoder.
320 */
321static void azione_funzionamento_valore(uint8_t identificatore,
322 const uint8_t array_stringa[])
323{
324 union float_bytes dato_valore1;
325 union float_bytes dato_valore2;
326
327 switch (identificatore)
328 {
329 /* Telegramma vuoto */
330 case 0x00U:
331 /* Non succede niente */
332 break;
333
334 /* Telegramma assegnazione velocità su encoder 1 */
335 case 0x01U:
336 (void) memcpy(dato_valore1.bytes, &array_stringa[0],
337 sizeof(float_t));
338 assegna_velocita_encoder1(dato_valore1.value);
339 break;
340
341 /* Telegramma assegnazione velocità su encoder 2 */
342 case 0x02U:
343 (void) memcpy(dato_valore2.bytes, &array_stringa[4],
344 sizeof(float_t));
345 assegna_velocita_encoder2(dato_valore2.value);
346 break;
347
348 /* Telegramma assegnazione velocità su entrambi gli encoder */
349 case 0x03U:
350 (void) memcpy(dato_valore1.bytes, &array_stringa[0],
351 sizeof(float_t));
352 (void) memcpy(dato_valore2.bytes, &array_stringa[4],
353 sizeof(float_t));
354 assegna_velocita_encoder1(dato_valore1.value);
355 assegna_velocita_encoder2(dato_valore2.value);
356 break;
357
358 /* Telegramma assegnazione accelerazione su encoder 1 */
359 case 0x04U:
360 (void) memcpy(dato_valore1.bytes, &array_stringa[0],
361 sizeof(float_t));
362 assegna_accelerazione_encoder1(dato_valore1.value);
363 break;
364
365 /* Telegramma assegnazione accelerazione su encoder 2 */
366 case 0x05U:
367 (void) memcpy(dato_valore2.bytes, &array_stringa[4],
368 sizeof(float_t));
369 assegna_accelerazione_encoder2(dato_valore2.value);
370 break;
371
372 /* Telegramma assegnazione accelerazione su entrambi gli encoder */
373 case 0x06U:
374 (void) memcpy(dato_valore1.bytes, &array_stringa[0],
375 sizeof(float_t));
376 (void) memcpy(dato_valore2.bytes, &array_stringa[4],
377 sizeof(float_t));
378 assegna_accelerazione_encoder1(dato_valore1.value);
379 assegna_accelerazione_encoder2(dato_valore2.value);
380 break;
381
382 /* Telegramma per disconnettersi dall'applicazione */
383 case 0x07U:
384 inizializza_variabili_encoder();
385 stato_connessione_app = false;
386 handshake_avvenuto = false;
387 break;
388
389 /* Telegramma per resettare la cinematica degli encoder */
390 case 0x08U:
391 assegna_accelerazione_encoder1(0);
392 assegna_accelerazione_encoder2(0);
393 assegna_velocita_encoder1(0);
394 assegna_velocita_encoder2(0);
395 break;
396
397 default:
398 /* Non succede niente */
399 break;
400 }
401}

431void manda_telegramma_di_risposta()
432{
433
434 if((stato_connessione_app == true) && (handshake_avvenuto == true))
435 {
436 uint8_t buffer[L_TELEGRAMMA_RISP]; /* 96 bits = 12 bytes */
437 union float_bytes temp_vel;
438 uint16_t temp_cont;
439
440 /* Mando velocita' del GIT 1*/
441 temp_vel.value = ritorna_velocita_encoder1();
442 (void) memcpy(buffer, temp_vel.bytes, sizeof(float_t));
443
444 /* Mando velocita' del GIT 2*/
445 temp_vel.value = ritorna_velocita_encoder2();
446 (void) memcpy(&buffer[4], temp_vel.bytes, sizeof(float_t));
447
448 /* Mando conteggio del GIT 1 */
449 temp_cont = ritorna_conteggio_encoder1();
450 buffer[8] = temp_cont & 0xFFU;
451 buffer[9] = (temp_cont >> 8U) & 0xFFU;
452
453 /* Mando conteggio del GIT 2 */
454 temp_cont = ritorna_conteggio_encoder2();
455 buffer[10] = temp_cont & 0xFFU;
456 buffer[11] = (temp_cont >> 8U) & 0xFFU;
457
458 // Send the buffer over UART
459 for(uint16_t indice = 0; indice < L_TELEGRAMMA_RISP; indice++)
460 {
461 XUartPs_SendByte(UART_BASEADDR, buffer[indice]);
462 }
463
464 handshake_avvenuto = false;
465 }
466 else
467 {
468 /* Non succede nulla */
469 }
470}
