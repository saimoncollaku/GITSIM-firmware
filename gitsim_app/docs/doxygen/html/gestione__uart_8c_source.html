51static void leggi_telegramma_di_connessione()
52{
53 uint16_t n_byte_ricevuti = 0;
54 uint8_t byte_ricevuti[L_TELEGRAMMA_CONN];
55 union float_bytes diametro;
56 union uint16_bytes ppr1;
57 union uint16_bytes ppr2;
58
59 do {
60 /* Attendo che un byte arrivi */
61 while (!(XUartPs_IsReceiveData(Uart_Ps.Config.BaseAddress)))
62 {
63 /* Wait */
64 }
65
66 /* Leggi byte ricevuto*/
67 uint8_t byte = XUartPs_ReadReg(Uart_Ps.Config.BaseAddress,
68 XUARTPS_FIFO_OFFSET) & 0xFF;
69
70 /* Salva byte ricevuto*/
71 byte_ricevuti[n_byte_ricevuti] = byte;
72 n_byte_ricevuti++;
73
74 }while(n_byte_ricevuti != L_TELEGRAMMA_CONN);
75
76
77 /* Estraggo diametro della ruota (little endian)*/
78 (void) memcpy(diametro.bytes, byte_ricevuti, sizeof(float_t));
79
80 /* Estraggo ppr encoder 1 (little endian)*/
81 (void) memcpy(ppr1.bytes, &byte_ricevuti[4], sizeof(uint16_t));
82
83 /* Estraggo ppr encoder 2 (little endian)*/
84 (void) memcpy(ppr2.bytes, &byte_ricevuti[6], sizeof(uint16_t));
85
86 /* Controllo se i parametri rientrano nei valori corretti */
87 if((diametro.value > MAX_DIAMETRO_RUOTA) || (diametro.value < MIN_DIAMETRO_RUOTA))
88 {
89 stato_connessione_app = false;
90 }
91 else if((ppr1.value > MAX_PPR_ENCODER) || (ppr1.value < MIN_PPR_ENCODER))
92 {
93 stato_connessione_app = false;
94 }
95 else if((ppr2.value > MAX_PPR_ENCODER) || (ppr2.value < MIN_PPR_ENCODER))
96 {
97 stato_connessione_app = false;
98 }
99 else
100 {
101 /* I controlli sono passiti, assegno i parametri agli encoder */
102 assegna_ppr_encoder1(ppr1.value);
103 assegna_ppr_encoder2(ppr2.value);
104 assegna_diametro_ruota(diametro.value);
105 aggiorna_passo_encoder1();
106 aggiorna_passo_encoder2();
107 stato_connessione_app = true;
108 }
109
110}

152static void azione_funzionamento_valore(uint8_t identificatore, const uint8_t array_stringa[])
153{
154 union float_bytes dato_valore1;
155 union float_bytes dato_valore2;
156
157 if(identificatore == 0x00U)
158 {
159 /* Non succede niente */
160 }
161 else if(identificatore == 0x01U)
162 {
163 (void) memcpy(dato_valore1.bytes, &array_stringa[0], sizeof(float_t));
164 assegna_velocita_encoder1(dato_valore1.value);
165 }
166 else if(identificatore == 0x02U)
167 {
168 (void) memcpy(dato_valore2.bytes, &array_stringa[4], sizeof(float_t));
169 assegna_velocita_encoder2(dato_valore2.value);
170 }
171 else if(identificatore == 0x03U)
172 {
173 (void) memcpy(dato_valore1.bytes, &array_stringa[0], sizeof(float_t));
174 (void) memcpy(dato_valore2.bytes, &array_stringa[4], sizeof(float_t));
175 assegna_velocita_encoder1(dato_valore1.value);
176 assegna_velocita_encoder2(dato_valore2.value);
177 }
178 else if(identificatore == 0x04U)
179 {
180 (void) memcpy(dato_valore1.bytes, &array_stringa[0], sizeof(float_t));
181 assegna_accelerazione_encoder1(dato_valore1.value);
182 }
183 else if(identificatore == 0x05U)
184 {
185 (void) memcpy(dato_valore2.bytes, &array_stringa[4], sizeof(float_t));
186 assegna_accelerazione_encoder2(dato_valore2.value);
187 }
188 else if(identificatore == 0x06U)
189 {
190 (void) memcpy(dato_valore1.bytes, &array_stringa[0], sizeof(float_t));
191 (void) memcpy(dato_valore2.bytes, &array_stringa[4], sizeof(float_t));
192 assegna_accelerazione_encoder1(dato_valore1.value);
193 assegna_accelerazione_encoder2(dato_valore2.value);
194 }
195 else if(identificatore == 0x07U)
196 {
197 inizializza_variabili_encoder();
198 stato_connessione_app = false;
199 handshake_avvenuto = false;
200 }
201 else if(identificatore == 0x08U)
202 {
203 assegna_accelerazione_encoder1(0);
204 assegna_accelerazione_encoder2(0);
205 assegna_velocita_encoder1(0);
206 assegna_velocita_encoder2(0);
207 }
208 else
209 {
210 /* Non succede niente */
211 }

244void manda_telegramma_di_risposta()
245{
246
247 if((stato_connessione_app == true) && (handshake_avvenuto == true))
248 {
249 uint8_t buffer[L_TELEGRAMMA_RISP]; /* 96 bits = 12 bytes */
250 union float_bytes temp_vel;
251 uint16_t temp_cont;
252
253 /* Mando velocita' del GIT 1*/
254 temp_vel.value = ritorna_velocita_encoder1();
255 (void) memcpy(buffer, temp_vel.bytes, sizeof(float_t));
256
257 /* Mando velocita' del GIT 2*/
258 temp_vel.value = ritorna_velocita_encoder2();
259 (void) memcpy(&buffer[4], temp_vel.bytes, sizeof(float_t));
260
261 /* Mando conteggio del GIT 1 */
262 temp_cont = ritorna_conteggio_encoder1();
263 buffer[8] = temp_cont & 0xFFU;
264 buffer[9] = (temp_cont >> 8U) & 0xFFU;
265
266 /* Mando conteggio del GIT 2 */
267 temp_cont = ritorna_conteggio_encoder2();
268 buffer[10] = temp_cont & 0xFFU;
269 buffer[11] = (temp_cont >> 8U) & 0xFFU;
270
271 // Send the buffer over UART
272 for(uint16_t indice = 0; indice < L_TELEGRAMMA_RISP; indice++)
273 {
274 XUartPs_SendByte(UART_BASEADDR, buffer[indice]);
275 }
276
277 handshake_avvenuto = false;
278 }
279 else
280 {
281 /* Non succede nulla */
282 }
283}
