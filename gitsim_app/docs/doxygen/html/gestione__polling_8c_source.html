54/**
55 * @brief Configura il sistema di interrupt GIC
56 *
57 * @param istanza_gic Puntatore all'istanza XScuGic da inizializzare
58 * @return s32 XST_SUCCESS se la configurazione ha successo, XST_FAILURE altrimenti
59 */
60static s32 configura_gic_system(XScuGic *istanza_gic)
61{
62 s32 status = XST_SUCCESS;
63
64 /* Configurazione dell'interrupt controller */
65 XScuGic_Config *gic_config;
66
67 /* Inizializzo il sistema di eccezioni */
68 Xil_ExceptionInit();
69
70 /* Ottengo la configurazione dell'interrupt controller */
71 gic_config = XScuGic_LookupConfig(INTC_DEVICE_ID);
72 if (NULL != gic_config)
73 {
74 /* Inizializza l'istanza dell'interrupt controller */
75 status = XScuGic_CfgInitialize(istanza_gic, gic_config,
76 gic_config->CpuBaseAddress);
77
78 /*
79 * Collega l'handler dell'interrupt alla logica di gestione di
80 * di interrupt hw del processore
81 */
82 Xil_ExceptionRegisterHandler(XIL_EXCEPTION_ID_INT,
83 (Xil_ExceptionHandler)XScuGic_InterruptHandler,
84 istanza_gic);
85
86 /* Abilita l'interrupt del processore */
87 Xil_ExceptionEnableMask(XIL_EXCEPTION_IRQ);
88 }
89 else
90 {
91 status = XST_FAILURE;
92 }
93
94 return status;
95}

98/**
99 * @brief Connette e configura l'interrupt per il timer SCU
100 *
101 * @param istanza_interrupt Puntatore all'istanza XScuGic (interrupt GIC)
102 * @param istanza_timer Puntatore all'istanza XScuTimer (timer SCU)
103 * @param n_timer_interrupt Numero dell'interrupt del timer
104 */
105static void connetti_interrupt_su_timer(XScuGic *istanza_interrupt,
106 XScuTimer *istanza_timer,
107 uint16_t n_timer_interrupt)
108{
109 /* Inizializzo il sistema di eccezioni */
110 Xil_ExceptionInit();
111
112 /* Associo il side loop al timer */
113 XScuGic_Connect(istanza_interrupt, n_timer_interrupt,
114 (Xil_ExceptionHandler)side_loop,
115 (void *)istanza_timer);
116
117 /* Abilito l'interrupt specifico per il GIC */
118 XScuGic_Enable(istanza_interrupt, n_timer_interrupt);
119
120 /* Abilito la generazione di interrupt nel timer SCU */
121 XScuTimer_EnableInterrupt(istanza_timer);
122
123 /* Abilito il sistema globale di gestione delle eccezioni */
124 Xil_ExceptionEnable();
125}

132/**
133 * @brief Inizializza e configura il timer per il polling
134 *
135 * Questa funzione configura l'interrupt GIC,
136 * inizializza il timer SCU, esegue un self-test, connette l'interrupt,
137 * e avvia il timer con le impostazioni specificate.
138 */
139void inizializza_polling_timer()
140{
141 s32 status = XST_SUCCESS;
142 XScuTimer_Config *scu_config_pointer;
143
144 /* Configura il Generic Interrupt Controller (GIC) */
145 status = configura_gic_system(&istanza_interrupt_gic);
146 while(status != XST_SUCCESS)
147 {
148 /* Configurazione interrupt fallita, mi blocco qui */
149 }
150
151 /* Inizializza il driver del timer SCU */
152 scu_config_pointer = XScuTimer_LookupConfig(TIMER_DEVICE_ID);
153 status = XScuTimer_CfgInitialize(&istanza_timer_scu, scu_config_pointer,
154 scu_config_pointer->BaseAddr);
155 while(status != XST_SUCCESS)
156 {
157 /* Configurazione timer fallita, mi blocco qui */
158 }
159
160 /* Eseguo un self-test del timer */
161 status = XScuTimer_SelfTest(&istanza_timer_scu);
162 while(status != XST_SUCCESS)
163 {
164 /* Self-test del timer fallito, mi blocco qui */
165 }
166
167 /* Configuro l'interrupt sul reset del timer */
168 connetti_interrupt_su_timer(&istanza_interrupt_gic, &istanza_timer_scu, TIMER_IRPT_INTR);
169
170 /* Configuro il timer per ripartire ad ogni overflow */
171 XScuTimer_EnableAutoReload(&istanza_timer_scu);
172
173 /* Imposto il prescaler del timer */
174 XScuTimer_SetPrescaler(&istanza_timer_scu, TIMER_PSC - 1U);
175
176 /* Imposto il valore di ricarica del timer */
177 XScuTimer_LoadTimer(&istanza_timer_scu, TIMER_LV - 1U);
178
179 /* Avvio il conteggio del timer */
180 XScuTimer_Start(&istanza_timer_scu);
181}

197/**
198 * @brief Calcola e restituisce il tempo di polling del timer
199 *
200 * Questa funzione calcola il tempo di polling basandosi sui valori di
201 * TIMER_PSC (prescaler), TIMER_LV (valore di caricamento), e APU_FREQ
202 * (frequenza dell'APU).
203 *
204 * Viene utilizzato nel side loop per dare il timing all'aggiornamento ed
205 * emulazione delle variabili di encoder.
206 *
207 * @return float_t Il tempo di polling calcolato in secondi
208 *
209 * @note La formula utilizzata e': t_polling = (TIMER_PSC * TIMER_LV) / (APU_FREQ * 0.5)
210 */
211float_t ritorna_tempo_del_polling(void)
212{
213 float_t t_polling = (TIMER_PSC * TIMER_LV) / (APU_FREQ * 0.5);
214 return t_polling;
215}
