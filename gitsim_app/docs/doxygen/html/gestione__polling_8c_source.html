46static s32 setup_interrupt_system(XScuGic *IstanzaGIC)
47{
48 s32 status = XST_SUCCESS;
49
50 XScuGic_Config *IntcConfig; /* Istanza dell'interrupt controller */
51
52 Xil_ExceptionInit();
53
54 /* Inizializza i driver dell'interrupt controller */
55 IntcConfig = XScuGic_LookupConfig(INTC_DEVICE_ID);
56 if (NULL != IntcConfig)
57 {
58 status = XScuGic_CfgInitialize(IstanzaGIC, IntcConfig,
59 IntcConfig->CpuBaseAddress);
60
61 /* Collega l'handler dell'interrupt alla logica di gestione di
62 * di interrupt hw del processore */
63 Xil_ExceptionRegisterHandler(XIL_EXCEPTION_ID_INT,
64 (Xil_ExceptionHandler)XScuGic_InterruptHandler,
65 IstanzaGIC);
66
67 /* Abilita l'interrupt del processore */
68 Xil_ExceptionEnableMask(XIL_EXCEPTION_IRQ);
69 }
70 else
71 {
72 status = XST_FAILURE;
73 }
74
75 return status;
76}

78static void connetti_interrupt_su_timer(XScuGic *istanza_interrupt,
79 XScuTimer *istanza_timer,
80 uint16_t ID_Timer_Interrupt)
81{
82 Xil_ExceptionInit();
83
84 /* Associazione funzione di interrupt */
85 XScuGic_Connect(istanza_interrupt, ID_Timer_Interrupt,
86 (Xil_ExceptionHandler)side,
87 (void *)istanza_timer);
88
89 /* Abilita GIC interrupt*/
90 XScuGic_Enable(istanza_interrupt, ID_Timer_Interrupt);
91
92 /* Abilita interrupt sul timer */
93 XScuTimer_EnableInterrupt(istanza_timer);
94 Xil_ExceptionEnable();
95}

101void inizializza_polling_timer()
102{
103 s32 status = XST_SUCCESS;
104 XScuTimer_Config *ConfigPtr;
105
106 /* Inizializza l'interrupt */
107 status = setup_interrupt_system(&istanza_interrupt);
108 while(status != XST_SUCCESS)
109 {
110 /* Interrupt setup fail */
111 }
112
113 /* Inizializza il driver del timer */
114 ConfigPtr = XScuTimer_LookupConfig(TIMER_DEVICE_ID);
115 status = XScuTimer_CfgInitialize(&istanza_timer_zynq, ConfigPtr,
116 ConfigPtr->BaseAddr);
117 while(status != XST_SUCCESS)
118 {
119 /* Timer setup fail */
120 }
121
122 /* Test */
123 status = XScuTimer_SelfTest(&istanza_timer_zynq);
124 while(status != XST_SUCCESS)
125 {
126 /* Timer selftest fail */
127 }
128
129 /* Connetti l'interrupt sul reset del timer */
130 connetti_interrupt_su_timer(&istanza_interrupt, &istanza_timer_zynq, TIMER_IRPT_INTR);
131
132 /* Abilita l'auto reset del timer */
133 XScuTimer_EnableAutoReload(&istanza_timer_zynq);
134
135 /* Imposta il prescaler */
136 XScuTimer_SetPrescaler(&istanza_timer_zynq, TIMER_PSC - 1U);
137
138 /* Carica valore a cui il timer si resetta */
139 XScuTimer_LoadTimer(&istanza_timer_zynq, TIMER_LV - 1U);
140
141 /* Inizia il conteggio */
142 XScuTimer_Start(&istanza_timer_zynq);
143}
