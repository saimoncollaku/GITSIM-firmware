27/** @brief Rappresenta lo stato dell'encoder basato sui livelli dei canali A e
28 * B.
29 *
30 * Utilizzato per decodificare la posizione e la generare il conteggio.
31 */
32typedef enum
33{
34 /** @brief Stato 0: Canale A = 0, Canale B = 0 */
35 zero,
36 /** @brief Stato 1: Canale A = 0, Canale B = 1 */
37 uno,
38 /** @brief Stato 2: Canale A = 1, Canale B = 0 */
39 due,
40 /** @brief Stato 3: Canale A = 1, Canale B = 1 */
41 tre,
42 /** @brief Stato incerto, usato per l'inizializzazione */
43 incerto
44}stato_encoder;

47/** @brief Struttura che rappresenta un encoder incrementale emulato.
48 *
49 * Questa struttura contiene tutti i parametri e le informazioni necessarie
50 * per emulare e gestire un encoder incrementale in un sistema embedded.
51 * Include caratteristiche fisiche, stati operativi/errore, e parametri di
52 * output.
53 */
54typedef struct
55{
56 /** @brief Duty cycle del segnale in uscita al canale A.
57 * Espresso come intero percentuale. Moltiplicare per 0.01 per ottenere il
58 * valore effettivo.
59 */
60 uint16_t duty_A;
61
62 /** @brief Duty cycle del segnale in uscita al canale B.
63 * Espresso come intero percentuale. Moltiplicare per 0.01 per ottenere il
64 * valore effettivo.
65 */
66 uint16_t duty_B;
67
68 /** @brief Flag per indicare l'incollaggio del canale A.
69 * true se il canale è incollato, false altrimenti.
70 */
71 bool incollaggio_A;
72
73 /** @brief Flag per indicare l'incollaggio del canale B.
74 * true se il canale è incollato, false altrimenti.
75 */
76 bool incollaggio_B;
77
78 /** @brief Flag per indicare errore di frequenza sul canale A.
79 * true se c'è un errore di frequenza, false altrimenti.
80 */
81 bool err_freq_A;
82
83 /** @brief Flag per indicare errore di frequenza sul canale B.
84 * true se c'è un errore di frequenza, false altrimenti.
85 */
86 bool err_freq_B;
87
88 /** @brief Errore di frequenza per passo.
89 * Rappresenta la deviazione dalla frequenza attesa per ogni passo dell'
90 * encoder.
91 */
92 float_t err_freq_passo;
93
94 /** @brief Sfasamento dei due segnali dell'encoder.
95 * Misurato in gradi.
96 */
97 int16_t fase;
98
99 /** @brief Impulsi per giro dell'encoder.
100 * Numero di suddivisioni del disco o impulsi per giro, misurato in PPR
101 * (Pulses Per Revolution).
102 */
103 uint16_t ppr;
104
105 /** @brief Diametro della ruota collegata al GIT.
106 * Misurato in metri.
107 */
108 float_t diametro;
109
110 /** @brief Lunghezza del passo di ruota rilevato da un canale dell'encoder.
111 * Misurato in metri. Ricalcolato quando diametro o risoluzione variano.
112 * Corrisponde alla lunghezza con risoluzione x2.
113 */
114 double_t l_passo;
115
116 /** @brief Posizione del sensore A.
117 * Misurato in metri. Rappresenta la posizione nell'intervallo di 4 passi,
118 * con risoluzione x2
119 */
120 double_t pos_A;
121
122 /** @brief Posizione del sensore B.
123 * Misurato in metri. Rappresenta la posizione nell'intervallo di 4 passi,
124 * con risoluzione x2
125 */
126 double_t pos_B;
127
128 /** @brief Velocità del GIT.
129 * Misurato in m/s.
130 */
131 double_t vel;
132
133 /** @brief Accelerazione del GIT.
134 * Misurato in m/s<sup>2</sup>
135 */
136 double_t acc;
137
138 /** @brief Numero di passi svolti dall'ultimo reset.
139 * Conteggio contato con i passi a risoluzione x4.
140 *
141 * @note quindi la lunghezza del passo equivalente per il
142 * calcolo dei conteggi e la metà rispetto a quello della
143 * variabile l_passo.
144 */
145 uint16_t conteggio;
146
147 /** @brief Indirizzo GPIO per il canale A.
148 * Struttura XGpio per la gestione del GPIO del canale A.
149 */
150 XGpio indirizzo_gpio_A;
151
152 /** @brief Indirizzo GPIO per il canale B.
153 * Struttura XGpio per la gestione del GPIO del canale B.
154 */
155 XGpio indirizzo_gpio_B;
156
157 /** @brief Stato corrente dell'encoder.
158 * Enumerazione che rappresenta lo stato operativo dell'encoder.
159 */
160 stato_encoder stato;
161
162} encoder;

400/**
401 * @brief Aggiorna il passo dell'encoder e_1
402 *
403 * Questa funzione calcola e aggiorna la lunghezza del passo dell'encoder e_1
404 * basandosi sul suo diametro e sul numero di impulsi per rivoluzione (ppr).
405 *
406 * La formula utilizzata è:
407 * @f[
408 * l_{passo} = \frac{diametro \cdot \pi}{ppr \cdot 2}
409 * @f]
410 * dove:
411 * - @f$l_{passo}@f$ è il passo dell'encoder
412 * - @f$diametro@f$ è il diametro della ruota dell'encoder
413 * - @f$\pi@f$ è il valore di PI greco
414 * - @f$ppr@f$ è il numero di impulsi per rivoluzione dell'encoder
415 * - Il fattore 2 nel denominatore è usato per ottenere la risoluzione x2 del
416 * passo
417 *
418 * @see e_1

420 */
